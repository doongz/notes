class Solution:

    def preorderTraversal(self, root: TreeNode) -> List[int]:

        # def dfs(node):

        # res = []

        # dfs(root)

        if not root: return []

        queue = [root]

        res = []

        while queue:

            node = queue.pop()

            res.append(node.val)

            if node.right:

                queue.append(node.right)

            if node.left:

                queue.append(node.left)

        return res
