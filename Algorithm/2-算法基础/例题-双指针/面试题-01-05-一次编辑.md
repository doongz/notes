题目：[面试题 01.05. 一次编辑](https://leetcode.cn/problems/one-away-lcci/)

字符串有三种编辑操作:插入一个英文字符、删除一个英文字符或者替换一个英文字符。 给定两个字符串，编写一个函数判定它们是否只需要一次(或者零次)编辑。

```
示例 1:

输入: 
first = "pale"
second = "ple"
输出: True
 

示例 2:

输入: 
first = "pales"
second = "pal"
输出: False

```



```c++
class Solution {
public:
    bool oneEditAway(string first, string second) {
        int n=first.size();
        int m = second.size();
        int i = 0;
        int j = 0;
        int flag = false;
        if (n+1==m) { // 插入    
            while (i<n&&j<m) {
                if (first[i]!=second[j]) {
                    if (flag) return false;
                    j++;
                    flag = true;
                } else {
                    i++;
                    j++;
                }
            }
        } else if (n-1 == m) { // 删除
            while (i<n&&j<m) {
                if (first[i]!=second[j]) {
                    if (flag) return false;
                    i++;
                    flag = true;
                } else {
                    i++;
                    j++;
                }
            }
        } else if (n==m) { // 改变
            while (i<n&&j<m) {
                if (first[i]!=second[j]) {
                    if (flag) return false;
                    i++;
                    j++;
                    flag = true;
                } else {
                    i++;
                    j++;
                }
            }
        } else {
            return false;
        }
        return true;
    }
};
```
